// Program.cs (ASP.NET Core 7+ minimal API)
var builder = WebApplication.CreateBuilder(args);
var services = builder.Services;
services.AddDbContext<AppDbContext>(opt => opt.UseSqlServer(builder.Configuration.GetConnectionString("Default")));
services.AddEndpointsApiExplorer();
services.AddSwaggerGen();

var app = builder.Build();
if (app.Environment.IsDevelopment()) { app.UseSwagger(); app.UseSwaggerUI(); }

app.MapGet("/tasks", async (AppDbContext db) => await db.Tasks.ToListAsync());
app.MapPost("/tasks", async (TaskItem t, AppDbContext db) => { db.Tasks.Add(t); await db.SaveChangesAsync(); return Results.Created($"/tasks/{t.Id}", t); });

app.Run();
